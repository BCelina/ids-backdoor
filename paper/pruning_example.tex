%\newcommand{\goodNodeStyle}{pattern=north west lines, pattern color=gray}
\newcommand{\goodNodeStyle}{fill=green}

\tikzset{
  my box/.style={color=#1},
%  align at top/.style={baseline=(current bounding box.north)},
}

\forestset{ 
  my depth label/.style={
    label={[my box=Fuchsia]right:#1},
  },
  my usages label/.style={
    label={[my box=MidnightBlue]left:#1},
  },
  default preamble={
    before typesetting nodes={
      !root.replace by={[, coordinate, append, name=veryTop]}
    },
    where n children>=1{
      diamond, aspect=2,
    }{},
    where level=0{}{
      if n=1{
        edge label={node[pos=.2, above] {Y}},
      }{
        edge label={node[pos=.2, above] {N}},
      }
    },
    for tree={
      edge+={thick, -Latex},
      math content,
      s sep+=2cm,
%      l sep=0cm,
      draw,
      thick,
      edge path'={ (!u) -| (.parent)},
    }
  }
}
